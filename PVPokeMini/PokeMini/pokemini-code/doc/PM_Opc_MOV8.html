Execute 

#nn     = Immediate unsigned 8-Bits
#ss     = Immediate signed 8-Bits
A       = Register A
B       = Register B
L       = Register L
H       = Register H
N       = Register N
F       = Register F
U/V     = Register U or V
I       = Register I
XI      = Register XI
YI      = Register YI
[N+#nn] = Memory: (I shl 16) or (N shl 8) or #nn
[HL]    = Memory: (I shl 16) or HL
[X]     = Memory: (XI shl 16) or X
[Y]     = Memory: (YI shl 16) or Y
[#nnnn] = Memory: #nnnn
[X+#ss] = Memory: (XI shl 16) or (X + #ss)
[Y+#ss] = Memory: (YI shl 16) or (Y + #ss)
[X+L]   = Memory: (XI shl 16) or (X + signed(L))
[Y+L]   = Memory: (YI shl 16) or (Y + signed(L))
    
; MOV Ds, Sc
;
; Ds = Destination
; Sc = Source

Ds = Sc

Description

8-Bits Source gets copied to the 8-Bits Destination.

Conditions

None

Examples 

; A = 0x55
MOV A, $80
; A = 0x80
    
; A = 0x12
; B = 0xCF
MOV B, A
; A = 0x12
; B = 0x12
    
; [HL] = 0xDE
; A = 0xCF
MOV A, [HL]
; [HL] = 0xDE
; A = 0xDE
